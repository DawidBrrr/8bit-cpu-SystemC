# Program: Test I/O ports
# Testign IO ports displaying

# load value into register A
A9 42    # LDA #0x42 (66)

# Save to I/O port 0 (display as a decimal number)
8D 00 FF # STA $FF00

# Load other value
A9 FF    # LDA #0xFF (255)

# Save to I/O port 1 (display as hex)  
8D 01 FF # STA $FF01

# Load ASCII code of letter 'A' (65)
A9 41    # LDA #0x41

# Load to I/O port 2 (display as character)
8D 02 FF # STA $FF02

# Load number for binary display
A9 55    # LDA #0x55 (01010101 w binarnym)

# Save to I/O port 3 (binary)
8D 03 FF # STA $FF03

# stop procesor
00       # BRK
